{ "permissions": { "allow": [ "Read", // broad: read anything "Write(src/**)", // scoped: source only "Write(tests/**)", // scoped: test files "Bash(npm test)", // exact: one command "Bash(npm run build)", // exact: one command "Bash(npx tsc *)", // scoped: tsc variants "Bash(git status)", // exact: read-only git "Bash(git diff *)" // scoped: diff variants ], "deny": [ "Bash(rm *)", // block all deletions "Bash(sudo *)", // block escalation "Bash(git push *)", // block push "Write(.env*)", // protect secrets "Write(config/prod*)" // protect prod config ] } }
